PYTHON PACKAGE:

  # Create models
  registry = ModelRegistry()
  model = registry.create("XGBoost", learning_rate=0.05)
  
  # Train
  model.fit(X_train, y_train)
  
  # Predict
  y_pred = model.predict(X_test)
  proba = model.predict_proba(X_test, horizons=[3, 6, 12])
  
  # Evaluate
  evaluator = BaselineEvaluator()
  metrics = evaluator.evaluate_model(y_test["event"], y_pred, times=y_test["time"])

================================================================================
DATA FORMAT SPECIFICATION
================================================================================

Input Format:
  X: pd.DataFrame or np.ndarray
     Shape: (n_samples, n_features)
     
  y: dict with keys:
     'event': np.ndarray of binary indicators (0/1)
     'time': np.ndarray of follow-up times (months)
     
  patient_ids: np.ndarray (optional)
     Patient identifiers for group-based splitting

Output Format:
  model.predict() returns np.ndarray [0, 1]
  model.predict_proba() returns dict {horizon: probabilities}
